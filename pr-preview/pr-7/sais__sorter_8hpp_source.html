81  size_type K) {

83  for (auto &v : S)
84  BA[v]++;

126 template <typename size_type = std::uint32_t>
127 void induceL(const std::ranges::random_access_range auto &S, size_type K,
128  const TypeVector &T, std::ranges::random_access_range auto &SA) {
129  auto n = (size_type)S.size();
130  auto BA = get_bucket<BUCKET_TYPE::BEGIN>(S, K);
131  for (auto i = size_type{0}; i < n; i++) {
132  if (SA[i] == EMPTY<size_type> or SA[i] == 0)
133  continue;
134 
135  auto j = SA[i] - 1;
136  if (T[j] != SUFFIX_TYPE::L_TYPE)
137  continue;
138 
139  SA[BA[S[j]]++] = j;
140  }

194 template <typename size_type = std::uint32_t>
195 auto is_same_substr(const std::ranges::random_access_range auto &S,
196  const TypeVector &T, size_type x, size_type y) {
197  auto n = (size_type)S.size();
198 
199  do {
200  if (x == n or y == n or S[x++] != S[y++])
201  return false;
202  } while (not is_LMS(T, x) and not is_LMS(T, y));
203 
204  return x != n and y != n and is_LMS(T, x) and is_LMS(T, y) and S[x] == S[y];

218 template <typename size_type = std::uint32_t>
219 auto name_lms_substr(const std::ranges::random_access_range auto &S,
220  size_type K, const TypeVector &T,
221  std::ranges::random_access_range auto &SA) {
222  auto n = (size_type)S.size();
223  auto n1 = size_type{};
224  for (auto i = size_type{0}; i <= n; i++)
225  if (is_LMS(T, SA[i]))
226  SA[n1++] = SA[i];
227 
228  std::fill(std::begin(SA) + n1, std::end(SA), EMPTY<size_type>);
229 
230  auto prev = EMPTY<size_type>, name_idx = size_type{};
231  for (auto i = size_type{1}; i < n1; i++) {
232  auto curr = SA[i];
233  name_idx +=
234  prev != EMPTY<size_type> and not is_same_substr(S, T, prev, curr);
235  SA[n1 + curr / 2] = name_idx;
236  prev = curr;
237  }
238 
239  for (auto i = n, j = n; i >= n1; i--)
240  if (SA[i] != EMPTY<size_type>)
241  SA[j--] = SA[i];
242 
243  // pop back tailing '\0'
244  n1--;
245 
246  return std::tuple<size_type, size_type>{n1, name_idx + 1};
247 }

293 template <typename size_type = std::uint32_t>
294 void suffix_array(const std::ranges::random_access_range auto &S, size_type K,
295  std::ranges::random_access_range auto &SA) {
296  // 1. get TypeVector from S
297  auto T = get_type<size_type>(S);
298 
299  // 2. put LMS character into SA in any order for each bucket
300  put_lms_substr(S, K, T, SA);
301 
302  // 3. induce LMS substring
303  induce_sort(S, K, T, SA);
304 
305  // 4. naming LMS substring
306  // |S1| = n1, |SA1| = n1 + 1
307  auto [n1, K1] = name_lms_substr(S, K, T, SA);
308  auto S1 = std::ranges::subrange(std::end(SA) - n1, std::end(SA));
309  auto SA1 = std::ranges::subrange(std::begin(SA), std::begin(SA) + n1 + 1);
310 
311  // 5. recursively solve LMS suffix
312  if (K1 < n1) {
313  suffix_array(S1, K1, SA1);
314  } else {
315  for (auto i = size_type{}; i < n1; i++)
316  SA1[S1[i] + 1] = i;
317  SA1[0] = n1;
318  }
319 
320  // 6. put LMS character into SA in the order of LMS suffix
321  put_lms_suffix(S, K, T, S1, SA, SA1);
322 
323  // 7. induce SA from LMS suffix
324  induce_sort(S, K, T, SA);
325 }
